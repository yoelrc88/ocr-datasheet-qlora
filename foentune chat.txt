April 10 2026:
Yes, this is possible-ish on a 12 GB RTX A2000, but only if you do it the lean way. olmOCR-2-7B-1025 is already a Qwen2.5-VL-7B-Instruct derivative that AllenAI further tuned with GRPO to improve math, tables, and other tricky OCR cases. AllenAI’s own card says the BF16 release is the one to use for further fine-tuning, while the FP8 release is what they recommend for practical deployment. 
So I would not fine-tune a Q8 inference artifact directly. Treat Q8 as a deployment format. The normal path is: start from the original checkpoint, load it quantized for training with 4-bit NF4 / QLoRA, train only adapters, then export and re-quantize afterward. That is the same pattern shown in Hugging Face’s VLM fine-tuning recipe. 
The catch is VRAM. Hugging Face describes 7B VLM fine-tuning as resource-intensive and uses an A100 in its cookbook example. On a 12 GB card, the realistic setup is batch size 1, aggressive memory saving, shorter sequences, smaller images, and probably not fine-tuning every module. Unsloth explicitly recommends dropping batch size to 1 or reducing sequence length when you OOM, and it lets you fine-tune only selected parts such as the vision layers, language layers, or attention / MLP layers. It also recommends keeping training images around 300–1000 px to avoid wasting memory.

What makes the runtime blow up
Three things:
Full-page training instead of cropped table regions

 If you train on whole datasheet pages, especially tall dense pages, runtime can easily go up 1.5–3× versus targeted table crops. That is because the visual tokens and text output are both heavier. This is consistent with the VLM recipes emphasizing that vision fine-tuning is compute-heavy and with Unsloth/Qwen-VL guidance exposing explicit image-size controls. 


Long output targets

 If your answer target is full markdown reconstruction of the page, not just the table JSON/HTML, training slows down. Decoder length matters a lot.


RL / GRPO instead of SFT

 Don’t. Hugging Face’s GRPO VLM recipe exists, but it is more expensive than plain SFT. For merged-cell repair, SFT is the sane first move.

def normalize_whitespace(text: str) -> str:
    text = text.replace("\r\n", "\n").replace("\r", "\n")
    text = re.sub(r"[ \t]+", " ", text)
    text = re.sub(r"\n\s*\n+", "\n", text)
    return text.strip()


def normalize_html(html: str) -> str:
    """
    Light HTML normalization so labels are consistent.
    Do NOT over-normalize and destroy structure.
    """
    html = normalize_whitespace(html)
    soup = BeautifulSoup(html, "html.parser")

    # Remove comments
    for comment in soup.find_all(string=lambda t: isinstance(t, type(soup.comment))):
        comment.extract()

    # Keep only one top-level table if labels are messy.
    table = soup.find("table")
    if table is None:
        raise ValueError("Label does not contain a <table> element.")

    # Optional cleanup: lowercase tag names are handled by parser output.
    normalized = str(table)
    normalized = normalized.replace("<br>", "<br/>")
    normalized = re.sub(r">\s+<", "><", normalized)
    return normalized.strip()

def read_jsonl(path: str, max_samples: int = None) -> List[Dict[str, Any]]:
    rows: List[Dict[str, Any]] = []
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            rows.append(json.loads(line))
    if max_samples is not None:
        rows = rows[:max_samples]
    return rows


def validate_and_prepare(rows: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    cleaned = []
    for i, row in enumerate(rows):
        sample_id = row.get("id", f"sample_{i:04d}")
        doc_id = row.get("doc_id", sample_id)
        image_path = row.get("image")
        html = row.get("html")

        if not image_path or not os.path.exists(image_path):
            raise FileNotFoundError(f"Missing image for {sample_id}: {image_path}")
        if not html:
            raise ValueError(f"Missing html label for {sample_id}")

        # Verify image opens.
        with Image.open(image_path) as img:
            img.verify()

        html = normalize_html(html)

        cleaned.append(
            {
                "id": sample_id,
                "doc_id": doc_id,
                "image": image_path,
                "html": html,
                "notes": row.get("notes", ""),
            }
        )
    return cleaned

def grouped_split(rows: List[Dict[str, Any]], seed: int = 42):
    ids = list(range(len(rows)))
    groups = [r["doc_id"] for r in rows]

    # 80/10/10 split. For tiny data, keep it simple and grouped.
    gss1 = GroupShuffleSplit(n_splits=1, train_size=0.8, random_state=seed)
    train_idx, temp_idx = next(gss1.split(ids, groups=groups))

    temp_rows = [rows[i] for i in temp_idx]
    temp_ids = list(range(len(temp_rows)))
    temp_groups = [r["doc_id"] for r in temp_rows]

    gss2 = GroupShuffleSplit(n_splits=1, train_size=0.5, random_state=seed + 1)
    val_subidx, test_subidx = next(gss2.split(temp_ids, groups=temp_groups))

    train_rows = [rows[i] for i in train_idx]
    val_rows = [temp_rows[i] for i in val_subidx]
    test_rows = [temp_rows[i] for i in test_subidx]
    return train_rows, val_rows, test_rows

processor = AutoProcessor.from_pretrained(PROCESSOR_ID)


def generate_text_from_sample(model, processor, sample, max_new_tokens=1024, device="cuda"):
    # Remove assistant turn for generation.
    inference_messages = sample["messages"][:2]
    text_input = processor.apply_chat_template(
        inference_messages,
        tokenize=False,
        add_generation_prompt=True,
    )
    image_inputs, video_inputs = process_vision_info(inference_messages)
    inputs = processor(
        text=[text_input],
        images=image_inputs,
        videos=video_inputs,
        return_tensors="pt",
    )
    inputs = {k: v.to(device) if hasattr(v, "to") else v for k, v in inputs.items()}

    with torch.inference_mode():
        generated_ids = model.generate(**inputs, max_new_tokens=max_new_tokens, do_sample=False)

    trimmed_ids = [out_ids[len(in_ids):] for in_ids, out_ids in zip(inputs["input_ids"], generated_ids)]
    output = processor.batch_decode(
        trimmed_ids,
        skip_special_tokens=True,
        clean_up_tokenization_spaces=False,
    )[0]
    return output

def structural_score(pred_html: str, gold_html: str) -> Dict[str, Any]:
    """
    Cheap structural checks. This is NOT a full metric, but enough for the first pass.
    """
    pred = normalize_html(pred_html)
    gold = normalize_html(gold_html)

    def count(pattern: str, text: str) -> int:
        return len(re.findall(pattern, text, flags=re.IGNORECASE))

    return {
        "exact_match": int(pred == gold),
        "pred_tr": count(r"<tr\b", pred),
        "gold_tr": count(r"<tr\b", gold),
        "pred_td_th": count(r"<(td|th)\b", pred),
        "gold_td_th": count(r"<(td|th)\b", gold),
        "pred_rowspan": count(r"rowspan=", pred),
        "gold_rowspan": count(r"rowspan=", gold),
        "pred_colspan": count(r"colspan=", pred),
        "gold_colspan": count(r"colspan=", gold),
    }
